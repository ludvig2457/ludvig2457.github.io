Professional Build System

Enterprise-grade build script with auto-detection and multiple architecture support.

build.sh 
#!/bin/bash
set -e

# Цвета для вывода
GREEN="\e[32m"
YELLOW="\e[33m"
RED="\e[31m"
CYAN="\e[36m"
RESET="\e[0m"

OS_NAME="LudvigOS"
BUILD_DIR="$PWD/build"
ISO_DIR="$PWD/iso"

# Выбор архитектуры
echo -e "${CYAN}Выбери архитектуру:${RESET}"
echo "1) 32-bit (i686) - BIOS"
echo "2) 64-bit (x86_64) - UEFI"
read -p "Введите выбор [1-2]: " ARCH_CHOICE

if [ "$ARCH_CHOICE" = "1" ]; then
    ARCH="i686-elf"
    BUILD_SUBDIR="i686"
    QEMU="qemu-system-i386"
    EFI_MODE=false
elif [ "$ARCH_CHOICE" = "2" ]; then
    ARCH="x86_64-elf"
    BUILD_SUBDIR="x86_64"
    QEMU="qemu-system-x86_64"
    EFI_MODE=true
else
    echo -e "${RED}Invalid choice! Defaulting to 32-bit.${RESET}"
    ARCH="i686-elf"
    BUILD_SUBDIR="i686"
    QEMU="qemu-system-i386"
    EFI_MODE=false
fi

echo -e "${GREEN}[*] Selected architecture: $ARCH${RESET}"
if [ "$EFI_MODE" = true ]; then
    echo -e "${GREEN}[*] Using UEFI boot${RESET}"
else
    echo -e "${GREEN}[*] Using BIOS boot${RESET}"
fi

# Основные шаги сборки
echo -e "${YELLOW}[1] Cleaning previous builds...${RESET}"
mkdir -p "$BUILD_DIR" "$ISO_DIR"
rm -rf "$BUILD_PATH" "$ISO_PATH"
mkdir -p "$BUILD_PATH/boot" "$BUILD_PATH/kernel" "$ISO_PATH/boot/grub"

# Компиляция ядра
echo -e "${YELLOW}[2] Compiling kernel...${RESET}"
"$ARCH-g++" -ffreestanding -O2 -Wall -Wextra -std=c++17 \
    -c kernel.cpp -o kernel.o

# Линковка
echo -e "${YELLOW}[3] Linking kernel...${RESET}"
"$ARCH-ld" -n -T linker.ld -o kernel.elf kernel.o

# Создание загрузочных файлов
echo -e "${YELLOW}[4] Creating boot files...${RESET}"
cp kernel.elf "$ISO_PATH/boot/"

# Конфигурация GRUB
echo -e "${YELLOW}[5] Configuring GRUB...${RESET}"
cat > "$ISO_PATH/boot/grub/grub.cfg" << EOF
set timeout=3
menuentry "LudvigOS" {
    multiboot /boot/kernel.elf
    boot
}
EOF

# Создание ISO
echo -e "${YELLOW}[6] Creating ISO image...${RESET}"
grub-mkrescue -o "${OS_NAME}-${BUILD_SUBDIR}.iso" "$ISO_PATH"

echo -e "${GREEN}[✓] Build completed!${RESET}"
echo -e "${GREEN}[✓] ISO: ${OS_NAME}-${BUILD_SUBDIR}.iso${RESET}"
